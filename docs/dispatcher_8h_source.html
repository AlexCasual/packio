128  template <typename F>
129  function_ptr_type wrap_sync(F&& fct)
130  {
131  using value_args =
132  internal::decay_tuple_t<typename internal::func_traits<F>::args_type>;
133  using result_type = typename internal::func_traits<F>::result_type;
134 
135  return std::make_shared<function_type>(
136  [fct = std::forward<F>(fct)](
137  completion_handler handler, const msgpack::object& args) mutable {
138  if (args.via.array.size != std::tuple_size_v<value_args>) {
139  // keep this check otherwise msgpack unpacker
140  // may silently drop arguments
141  PACKIO_DEBUG("incompatible argument count");
142  handler.set_error("Incompatible arguments");
143  return;
144  }
145 
146  try {
147  if constexpr (std::is_void_v<result_type>) {
148  std::apply(fct, args.as<value_args>());
149  handler();
150  }
151  else {
152  handler(std::apply(fct, args.as<value_args>()));
153  }
154  }
155  catch (msgpack::type_error&) {
156  PACKIO_DEBUG("incompatible arguments");
157  handler.set_error("Incompatible arguments");
158  }
159  });
160  }

166  using value_args = internal::decay_tuple_t<internal::shift_tuple_t<args>>;
167 
168  return std::make_shared<function_type>(
169  [fct = std::forward<F>(fct)](
170  completion_handler handler, const msgpack::object& args) mutable {
171  if (args.via.array.size != std::tuple_size_v<value_args>) {
172  // keep this check otherwise msgpack unpacker
173  // may silently drop arguments
174  PACKIO_DEBUG("incompatible argument count");
175  handler.set_error("Incompatible arguments");
176  return;
177  }
178 
179  try {
180  std::apply(
181  [&](auto&&... args) {
182  fct(std::move(handler),
183  std::forward<decltype(args)>(args)...);
184  },
185  args.as<value_args>());
186  }
187  catch (msgpack::type_error&) {
188  PACKIO_DEBUG("incompatible arguments");
189  handler.set_error("Incompatible arguments");
190  }
191  });
192  }
