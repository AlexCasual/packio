81  void async_read(std::unique_ptr<msgpack::unpacker> unpacker)
82  {
83  // abort R/W on error
84  if (!socket_.is_open()) {
85  return;
86  }
87 
88  unpacker->reserve_buffer(buffer_reserve_size_);
89  auto buffer = boost::asio::buffer(
90  unpacker->buffer(), unpacker->buffer_capacity());
91  socket_.async_read_some(
92  buffer,
93  [self = shared_from_this(), unpacker = std::move(unpacker)](
94  boost::system::error_code ec, size_t length) mutable {
95  if (ec) {
96  PACKIO_WARN("read error: {}", ec.message());
97  self->close_connection();
98  return;
99  }
100 
101  PACKIO_TRACE("read: {}", length);
102  unpacker->buffer_consumed(length);
103 
104  for (msgpack::object_handle call; unpacker->next(call);) {
105  // handle the call asynchronously (post)
106  // to schedule the next read immediately
107  // this will allow parallel call handling
108  // in multi-threaded environments
109  boost::asio::post(
110  self->get_executor(), [self, call = std::move(call)] {
111  self->dispatch(call.get());
112  });
113  }
114 
115  self->async_read(std::move(unpacker));
116  });
117  }
118 
119  void dispatch(const msgpack::object& msgpack_call)
120  {
121  std::optional<Call> call = parse_call(msgpack_call);
122  if (!call) {
123  close_connection();
124  return;
125  }
126 
127  auto completion_handler =
128  [type = call->type, id = call->id, self = shared_from_this()](
129  boost::system::error_code ec, msgpack::object_handle result) {
130  if (type == msgpack_rpc_type::request) {
131  PACKIO_TRACE("result: {}", ec.message());
132  self->async_send_result(id, ec, std::move(result));
133  }
134  };
135 
136  const auto function = dispatcher_ptr_->get(call->name);
137  if (function) {
138  PACKIO_TRACE("call: {} (id={})", call->name, call->id);
139  (*function)(completion_handler, call->args);
140  }
141  else {
142  PACKIO_DEBUG("unknown function {}", call->name);
143  completion_handler(make_error_code(error::unknown_procedure), {});
144  }
145  }
146 
147  std::optional<Call> parse_call(const msgpack::object& call)
148  {
149  if (call.type != msgpack::type::ARRAY || call.via.array.size < 3) {
150  PACKIO_ERROR("unexpected message type: {}", call.type);
151  return std::nullopt;
152  }
153 
154  try {
155  int idx = 0;
156  id_type id = 0;
157  msgpack_rpc_type type = static_cast<msgpack_rpc_type>(
158  call.via.array.ptr[idx++].as<int>());
159 
160  std::size_t expected_size;
161  switch (type) {
162  case msgpack_rpc_type::request:
163  id = call.via.array.ptr[idx++].as<id_type>();
164  expected_size = 4;
165  break;
166  case msgpack_rpc_type::notification:
167  expected_size = 3;
168  break;
169  default:
170  PACKIO_ERROR("unexpected type: {}", type);
171  return std::nullopt;
172  }
173 
174  if (call.via.array.size != expected_size) {
175  PACKIO_ERROR("unexpected message size: {}", call.via.array.size);
176  return std::nullopt;
177  }
178 
179  std::string name = call.via.array.ptr[idx++].as<std::string>();
180  const msgpack::object& args = call.via.array.ptr[idx++];
181 
182  return Call{type, id, name, args};
183  }
184  catch (msgpack::type_error& exc) {
185  PACKIO_ERROR("unexpected message content: {}", exc.what());
186  (void)exc;
187  return std::nullopt;
188  }
189  }

191  void async_send_result(
192  id_type id,
193  boost::system::error_code ec,
194  msgpack::object_handle result_handle)
195  {
196  // abort R/W on error
197  if (!socket_.is_open()) {
198  return;
199  }
200 
201  auto message_ptr = std::make_unique<message_type>();
202  msgpack::packer<buffer_type> packer(std::get<buffer_type>(*message_ptr));
203 
204  // serialize the result into the buffer
205  const auto pack = [&](auto&& error, auto&& result) {
206  packer.pack(std::forward_as_tuple(
207  static_cast<int>(msgpack_rpc_type::response),
208  id,
209  std::forward<decltype(error)>(error),
210  std::forward<decltype(result)>(result)));
211  };
212 
213  if (ec) {
214  if (result_handle.get().is_nil()) {
215  pack(ec.message(), msgpack::type::nil_t{});
216  }
217  else {
218  pack(result_handle.get(), msgpack::type::nil_t{});
219  }
220  }
221  else {
222  pack(msgpack::type::nil_t{}, result_handle.get());
223  }
224 
225  // move the result handle to the message pointer
226  // as the buffer is non-owning, we need to keep the result handle
227  // with the buffer
228  std::get<msgpack::object_handle>(*message_ptr) = std::move(result_handle);
229  async_send_message(std::move(message_ptr));
230  }
231 
232  void async_send_message(std::unique_ptr<message_type> message_ptr)
233  {
234  wstrand_.push([this,
235  self = shared_from_this(),
236  message_ptr = std::move(message_ptr)]() mutable {
237  using internal::buffer;
238 
239  auto buf = buffer(std::get<buffer_type>(*message_ptr));
240  boost::asio::async_write(
241  socket_,
242  buf,
243  [self = std::move(self), message_ptr = std::move(message_ptr)](
244  boost::system::error_code ec, size_t length) {
245  self->wstrand_.next();
246 
247  if (ec) {
248  PACKIO_WARN("write error: {}", ec.message());
249  self->close_connection();
250  return;
251  }
252 
253  PACKIO_TRACE("write: {}", length);
254  (void)length;
255  });
256  });
257  };
258 
259  void close_connection()
260  {
261  boost::system::error_code ec;
262  socket_.close(ec);
263  if (ec) {
264  PACKIO_WARN("close error: {}", ec.message());
265  }
266  }
